# Low-confidence secret patterns (PCRE/Python regex) — match produces WARN only (never block)
# Based on variable naming conventions. These patterns have higher false-positive
# rates (test values, placeholders, example strings) and must not block writes.
# Each line is one PCRE-style/Python `re`-compatible regular expression. Blank
# lines and '#' lines are ignored.

# Variable named 'password' assigned a non-empty non-placeholder string
(?i)\bpassword\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'token' assigned a string
(?i)\btoken\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'secret' assigned a string
(?i)\bsecret\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'api_key' or 'apikey' assigned a string
(?i)\bapi_?key\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'auth_token' assigned a string
(?i)\bauth_token\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'access_key' assigned a string
(?i)\baccess_key\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'private_key' assigned a string (not PEM)
(?i)\bprivate_key\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'client_secret' assigned a string
(?i)\bclient_secret\s*[=:]\s*['"][^'"]{4,}['"]

# Variable named 'database_url' or 'db_url' containing credentials
(?i)\b(database_url|db_url)\s*[=:]\s*['"][^'"]*:[^'"]*@[^'"]+['"]

# Variable named 'connection_string' containing credentials
(?i)\bconnection_string\s*[=:]\s*['"][^'"]*password[^'"]*['"]
