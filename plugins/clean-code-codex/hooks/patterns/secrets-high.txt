# High-confidence secret patterns (PCRE/Python-style regex) — match justifies
# blocking on Claude Code
# Each line is one regular expression using syntax compatible with `grep -P`
# and Python `re`. Blank lines and lines starting with '#' are ignored by
# hook-lib.sh.

# AWS Access Key ID
AKIA[0-9A-Z]{16}

# AWS Secret Access Key (40-char hex/alphanumeric after common key names)
(?i)(aws_secret_access_key|aws_secret)\s*[=:]\s*['"]?[A-Za-z0-9/+]{40}['"]?

# GitHub Personal Access Token (classic)
ghp_[A-Za-z0-9]{36}

# GitHub Fine-grained PAT
github_pat_[A-Za-z0-9_]{82}

# GitHub Actions token
ghs_[A-Za-z0-9]{36}

# GitHub OAuth token
gho_[A-Za-z0-9]{36}

# Slack Bot/App token
xox[baprs]-[A-Za-z0-9-]{10,}

# Slack Webhook URL
https://hooks\.slack\.com/services/[A-Za-z0-9/]+

# Stripe secret key
sk_live_[A-Za-z0-9]{24,}

# Stripe restricted key
rk_live_[A-Za-z0-9]{24,}

# Twilio Account SID + Auth Token pattern
AC[a-f0-9]{32}

# SendGrid API key
SG\.[A-Za-z0-9_-]{22}\.[A-Za-z0-9_-]{43}

# Mailgun API key
key-[a-f0-9]{32}

# PEM private key header
-----BEGIN (RSA |EC |DSA |OPENSSH |PRIVATE |ENCRYPTED )?PRIVATE KEY-----

# PEM certificate
-----BEGIN CERTIFICATE-----

# Generic high-entropy secret: 32+ hex chars in assignment context
(?i)(api_key|apikey|auth_token|access_token|secret_key|private_key)\s*[=:]\s*['"]?[a-f0-9]{32,}['"]?

# Generic high-entropy base64-ish string (40+ chars) in assignment context
(?i)(api_key|apikey|auth_token|access_token|secret_key|private_key|bearer)\s*[=:]\s*['"][A-Za-z0-9+/]{40,}={0,2}['"]

# npm auth token in .npmrc
//registry\.npmjs\.org/:_authToken\s*=\s*\S+

# Generic JWT (three base64url segments)
eyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}
